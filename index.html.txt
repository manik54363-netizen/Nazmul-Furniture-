<script>

let products = JSON.parse(localStorage.getItem("products")) || [];

function adminLogin(){
    let pass = prompt("অ্যাডমিন পাসওয়ার্ড দিন:");
    if(pass === "1234"){
        document.getElementById("adminPanel").style.display="block";
    }else{
        alert("ভুল পাসওয়ার্ড!");
    }
}

function displayProducts(){
    let list = document.getElementById("productList");
    list.innerHTML="";
    products.forEach((p,index)=>{
        list.innerHTML += `
        <div class="card">
            <img src="${p.image}">
            <h3>${p.name}</h3>

            <a href="https://wa.me/8801XXXXXXXXX?text=আমি ${p.name} অর্ডার করতে চাই" target="_blank">
                <button>WhatsApp অর্ডার</button>
            </a>

            <button onclick="alert('বিকাশ করুন: 017XXXXXXXX')">বিকাশ পেমেন্ট</button>

            <button onclick="deleteProduct(${index})">ডিলিট</button>
        </div>
        `;
    });
}

function addProduct(){
    let name = document.getElementById("pname").value;
    let file = document.getElementById("pimage").files[0];

    if(name === "" || !file){
        alert("সব তথ্য দিন");
        return;
    }

    let reader = new FileReader();
    reader.onload = function(){
        products.push({name:name, image:reader.result});
        localStorage.setItem("products", JSON.stringify(products));
        displayProducts();
    }
    reader.readAsDataURL(file);
}

function deleteProduct(index){
    products.splice(index,1);
    localStorage.setItem("products", JSON.stringify(products));
    displayProducts();
}

displayProducts();

</script>
